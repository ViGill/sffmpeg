PROJECT(sffmpeg)

CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
INCLUDE(ExternalProject)
INCLUDE_DIRECTORIES(BEFORE ${CMAKE_BINARY_DIR}/include)
LINK_DIRECTORIES(${CMAKE_BINARY_DIR}/lib)
SET(ENV{PATH} ${CMAKE_BINARY_DIR}/bin:$ENV{PATH})
SET(ENV{PKG_CONFIG_PATH} ${CMAKE_BINARY_DIR}/lib/pkgconfig)


IF(${CROSS_ARCH} MATCHES "^x86$")
    MESSAGE("Building sffmpeg for x86")
    INCLUDE("CMakeLists-x86.txt")
ELSEIF(${CROSS_ARCH} MATCHES "^armv7$")
    MESSAGE("Building sffmpeg for ARM V7")
    INCLUDE("CMakeLists-armv7.txt")
ELSE()
    MESSAGE("Invalid Arch " [${CMAKE_ARCH}])
ENDIF()
